public class Solution {
    public boolean canJump(int[] nums) {
        int reachable = 0;  // 현재까지 도달 가능한 최대 인덱스

        for (int i = 0; i < nums.length; i++) {
            if (i > reachable) return false;  // 현재 위치가 도달 불가능하면 종료
            reachable = Math.max(reachable, i + nums[i]);
        }

        return true;
    }
}
