public class Solution {
    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total = 0;  // 전체적으로 남는 연료
        int tank = 0;   // 현재 시점까지 남은 연료
        int start = 0;  // 시작 인덱스 후보

        for (int i = 0; i < gas.length; i++) {
            int diff = gas[i] - cost[i];
            total += diff;
            tank += diff;

            if (tank < 0) {
                // 이 시점까지는 못 감 → 다음 정거장을 시작점으로 변경
                start = i + 1;
                tank = 0;
            }
        }

        return total >= 0 ? start : -1;
    }
}
